CREATE TABLE hr_staff_salary (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  staff_id UUID REFERENCES staff(id) UNIQUE,
  basic_salary DECIMAL(10,2) NOT NULL,
  epf_employee_rate DECIMAL(5,2) DEFAULT 11,    -- % pekerja
  epf_employer_rate DECIMAL(5,2) DEFAULT 13,    -- % majikan
  socso_category INT DEFAULT 1,
  eis_enabled BOOLEAN DEFAULT true,
  tax_category TEXT DEFAULT 'single',            -- single/married
  pcb_enabled BOOLEAN DEFAULT true,
  bank_name TEXT,
  bank_account TEXT,
  effective_date DATE DEFAULT CURRENT_DATE,
  created_at TIMESTAMPTZ DEFAULT now()
);
B. Statutory rates config table:

CREATE TABLE hr_statutory_rates (
  id TEXT PRIMARY KEY,          -- e.g. 'epf_2025', 'socso_2025'

  year INT NOT NULL,
  brackets JSONB NOT NULL,      -- rate brackets by salary range
  is_active BOOLEAN DEFAULT true
);

Fasa 2: Leave Management + 2-Layer Approval
A. Tables:

CREATE TABLE hr_leave_types (
  id TEXT PRIMARY KEY,
  label TEXT NOT NULL,          -- "Annual Leave"
  short_code TEXT NOT NULL,     -- "AL"
  default_days INT DEFAULT 0,  -- entitlement per year
  is_paid BOOLEAN DEFAULT true,
  requires_doc BOOLEAN DEFAULT false,  -- MC needs document
  color TEXT DEFAULT '#3b82f6',
  is_active BOOLEAN DEFAULT true,
  sort_order INT DEFAULT 0
);

CREATE TABLE hr_leave_entitlements (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  staff_id UUID REFERENCES staff(id),
  leave_type_id TEXT REFERENCES hr_leave_types(id),
  year INT NOT NULL,
  total_days INT NOT NULL,
  used_days DECIMAL(4,1) DEFAULT 0,
  carried_over INT DEFAULT 0,
  UNIQUE(staff_id, leave_type_id, year)
);

CREATE TABLE hr_leave_requests (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  staff_id UUID REFERENCES staff(id),
  leave_type_id TEXT REFERENCES hr_leave_types(id),
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  total_days DECIMAL(4,1) NOT NULL,
  reason TEXT,
  attachment_url TEXT,
  status TEXT DEFAULT 'pending',          -- pending → layer1_approved → approved / rejected
  layer1_approver_id UUID,
  layer1_approved_at TIMESTAMPTZ,
  layer1_note TEXT,
  layer2_approver_id UUID,
  layer2_approved_at TIMESTAMPTZ,
  layer2_note TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

Fasa 3: Attendance Performance Scoring + Dashboard
A. Performance calculation:

Score formula (customizable by HR):
Hadir tepat masa: +10 pts
Lewat < 15min: +5 pts
Lewat > 15min: 0 pts
Tidak hadir tanpa cuti: -10 pts
MC: neutral (0 pts)
Monthly score = total points / working days × 100
Store dalam hr_performance_scores table
B. Dashboard:

Staff card: current month score, trend chart (3 months)
HR view: ranking semua staff, filter by department/branch
Color coding: 90+ hijau, 70-89 kuning, <70 merah
C. Integration with payroll:

Optional: attendance bonus/deduction based on score threshold
